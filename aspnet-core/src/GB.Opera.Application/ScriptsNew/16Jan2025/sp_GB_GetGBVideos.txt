ALTER Procedure [dbo].[sp_GB_GetGBVideos]
@type Int
AS
BEGIN


	IF (@type =1)
	  BEGIN

	  SELECT * from GulfBaseVideos WHERE sortorder >= 1 AND sortorder <= 2 Order by Id DESC

	  END
	ELSE 
	  BEGIN
		
		SELECT * from GulfBaseVideos WHERE sortorder >= 3 AND sortorder <= 4 Order by Id DESC

	  END

END
